# HiFiBerry Digi+ Configuration for Raspberry Pi 4 with 4K HDMI TV
# Copy this to /boot/firmware/config.txt or /boot/config.txt

# GPU Memory (increase for 4K video)
gpu_mem=512

# HiFiBerry Digi+ overlay (S/PDIF output)
dtoverlay=hifiberry-digi

# Disable onboard audio
dtparam=audio=off

# 4K HDMI settings
hdmi_enable_4kp60=1
hdmi_force_hotplug=1
hdmi_group=2
hdmi_mode=97  # 3840x2160 @ 30Hz (use 107 for 60Hz if supported)

# HDMI audio off (using HiFiBerry Digi+ S/PDIF)
hdmi_drive=1

# Enable hardware video acceleration
dtoverlay=vc4-kms-v3d
max_framebuffers=2

# Overclock settings for smooth 4K (optional, use with caution)
# over_voltage=6
# arm_freq=2147

# Boot options
disable_splash=1
boot_delay=0
